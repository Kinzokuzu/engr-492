def chunk_data(ds, n=20):
    """
    Returns a the data set (ds) subdivided by n points.
    """
    ds_list = []
    chunk = []
    for i in range(len(ds)):
        chunk.append(ds[i])
        if i % n == n-1:
            ds_list.append(chunk)
            chunk.clear()

    return ds_list

import math
import statistics as stats

def mean(data):
    """
    Returns the mean magnitude of a list of values.
    """
    return stats.mean( [abs(d) for d in data] )
    

def rms(data):
    """
    Returns the root mean square of a list of values.
    """
    return math.sqrt(stats.mean( [d ** 2 for d in data] ))
    

def slope_changes(data):
    """
    Return the number of times the data's slope changes form
    positive to negative or visa versa.
    """
    change_count = 0
    prev_slope = (data[1] - data[0]) > 0
    for i in range(2, len(data)):
        curr_slope = (data[i] - data[i-1]) > 0
        if curr_slope != prev_slope:
            prev_slope = curr_slope
            change_count += 1

    return change_count
            

def pos_peaks(data):
    """
    Returns the number of local maxima in the data set.
    """
    peak_count = 0
    for i in range(1, len(data)-1):
        if (data[i] > data[i-1]) and (data[i] > data[i+1]):
            peak_count += 1

    return peak_count
    

def neg_peaks(data):
    """
    Returns the number of local minima in the data set.
    """
    peak_count = 0
    for i in range(1, len(data)-1):
        if (data[i] < data[i-1]) and (data[i] < data[i+1]):
            peak_count += 1

    return peak_count
    

def zero_crossings(data):
    """
    Return the number of times the data set crosses the x-axis.
    """
    crossing_count = 0
    for i in range(1, len(data)):
        if data[i] > 0 and data[i-1] < 0:
            crossing_count += 1
        elif data[i] < 0 and data[i-1] > 0:
            crossing_count += 1

    return crossing_count
    

def extract_features(data):
    """
    Returns a list of tuples containing the extracted features (see above)
    of each chunk in a data set.
    """
    feature_list = []
    for d in data:
        feature_list.append( (mean(d), rms(d), slope_changes(d), pos_peaks(d), neg_peaks(d), zero_crossings(d)) )

    return feature_list
